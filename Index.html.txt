<script crossorigin src="https://unpkg.com/react@18/umd/react.production.min.js"></script>

<script type="text/babel">
    const { useState, useEffect } = React;

    const BestDressAward = () => {
        const [isAdmin, setIsAdmin] = useState(false);
        const [adminPassword, setAdminPassword] = useState('');
        const [isVotingOpen, setIsVotingOpen] = useState(true);
        const [maleVote, setMaleVote] = useState(null);
        const [femaleVote, setFemaleVote] = useState(null);
        const [showAdminLogin, setShowAdminLogin] = useState(false);
        
        const [userId] = useState(() => {
            let id = localStorage.getItem('bestdress-user-id');
            if (!id) {
                id = 'user-' + Date.now() + '-' + Math.random().toString(36).substr(2, 9);
                localStorage.setItem('bestdress-user-id', id);
            }
            return id;
        });

        const maleColleagues = [
            'Aneez', 'Ian', 'Joshua', 'Hui Heng', 'Soon Hau', 'Ebenezer',
            'Ravi', 'Ray', 'Kong', 'Jason', 'KY', 'Wilson Teh',
            'Jaden', 'Samuel', 'Rodney', 'Vimal'
        ];

        const femaleColleagues = [
            'Sarah', 'Shixian', 'Emily', 'Christine', 'Pallavi', 'Megha',
            'Pearlyn', 'Li Xuan', 'Michelle', 'Kelly', 'Syafika', 'Su Lyn', 'Chelvi'
        ];

        const [maleVotes, setMaleVotes] = useState({});
        const [femaleVotes, setFemaleVotes] = useState({});

        useEffect(() => {
            loadData();
        }, []);

        const loadData = () => {
            try {
                const votingStatus = localStorage.getItem('bestdress-voting-open');
                const maleVotesData = localStorage.getItem('bestdress-male-votes');
                const femaleVotesData = localStorage.getItem('bestdress-female-votes');
                const userMaleVote = localStorage.getItem(`bestdress-male-vote-${userId}`);
                const userFemaleVote = localStorage.getItem(`bestdress-female-vote-${userId}`);
                
                if (votingStatus !== null) {
                    setIsVotingOpen(votingStatus === 'true');
                }
                if (maleVotesData) {
                    setMaleVotes(JSON.parse(maleVotesData));
                }
                if (femaleVotesData) {
                    setFemaleVotes(JSON.parse(femaleVotesData));
                }
                if (userMaleVote) {
                    setMaleVote(userMaleVote);
                }
                if (userFemaleVote) {
                    setFemaleVote(userFemaleVote);
                }
            } catch (error) {
                console.log('No existing data found, starting fresh');
            }
        };

        const handleAdminLogin = () => {
            if (adminPassword === 'admin123') {
                setIsAdmin(true);
                setShowAdminLogin(false);
                setAdminPassword('');
            } else {
                alert('Incorrect password');
            }
        };

        const toggleVoting = () => {
            const newStatus = !isVotingOpen;
            localStorage.setItem('bestdress-voting-open', newStatus.toString());
            setIsVotingOpen(newStatus);
        };

        const handleMaleVote = (name) => {
            if (!isVotingOpen) {
                alert('Voting is currently closed');
                return;
            }

            let updatedVotes = { ...maleVotes };
            if (maleVote) {
                updatedVotes[maleVote] = Math.max(0, (updatedVotes[maleVote] || 0) - 1);
            }
            updatedVotes[name] = (updatedVotes[name] || 0) + 1;

            localStorage.setItem('bestdress-male-votes', JSON.stringify(updatedVotes));
            localStorage.setItem(`bestdress-male-vote-${userId}`, name);
            setMaleVotes(updatedVotes);
            setMaleVote(name);
        };

        const handleFemaleVote = (name) => {
            if (!isVotingOpen) {
                alert('Voting is currently closed');
                return;
            }

            let updatedVotes = { ...femaleVotes };
            if (femaleVote) {
                updatedVotes[femaleVote] = Math.max(0, (updatedVotes[femaleVote] || 0) - 1);
            }
            updatedVotes[name] = (updatedVotes[name] || 0) + 1;

            localStorage.setItem('bestdress-female-votes', JSON.stringify(updatedVotes));
            localStorage.setItem(`bestdress-female-vote-${userId}`, name);
            setFemaleVotes(updatedVotes);
            setFemaleVote(name);
        };

        const getMaleWinner = () => {
            if (Object.keys(maleVotes).length === 0) return null;
            return Object.entries(maleVotes).reduce((a, b) => a[1] > b[1] ? a : b)[0];
        };

        const getFemaleWinner = () => {
            if (Object.keys(femaleVotes).length === 0) return null;
            return Object.entries(femaleVotes).reduce((a, b) => a[1] > b[1] ? a : b)[0];
        };

        const totalMaleVotes = Object.values(maleVotes).reduce((sum, v) => sum + v, 0);
        const totalFemaleVotes = Object.values(femaleVotes).reduce((sum, v) => sum + v, 0);

        const getSortedVotes = (votes) => {
            return Object.entries(votes)
                .sort((a, b) => b[1] - a[1])
                .filter(([_, count]) => count > 0);
        };

        return (
            <div className="max-w-6xl mx-auto">
                {/* Header */}
                <div className="bg-white rounded-lg shadow-2xl p-6 mb-6">
                    <div className="flex items-center justify-between">
                        <div className="flex items-center gap-3">
                            <svg className="w-10 h-10 text-yellow-500" fill="currentColor" viewBox="0 0 24 24">
                                <path d="M12 2L15.09 8.26L22 9.27L17 14.14L18.18 21.02L12 17.77L5.82 21.02L7 14.14L2 9.27L8.91 8.26L12 2Z"/>
                            </svg>
                            <div>
                                <h1 className="text-3xl font-bold text-gray-800">Best Dress Award</h1>
                                <p className="text-gray-600">Vote for the Best Dressed Male & Female Colleague</p>
                            </div>
                        </div>
                        <button
                            onClick={() => setShowAdminLogin(!showAdminLogin)}
                            className="px-4 py-2 bg-gray-800 text-white rounded-lg hover:bg-gray-700 transition-colors flex items-center gap-2"
                        >
                            <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                {isAdmin ? (
                                    <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M8 11V7a4 4 0 118 0m-4 8v2m-6 4h12a2 2 0 002-2v-6a2 2 0 00-2-2H6a2 2 0 00-2 2v6a2 2 0 002 2z" />
                                ) : (
                                    <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M12 15v2m-6 4h12a2 2 0 002-2v-6a2 2 0 00-2-2H6a2 2 0 00-2 2v6a2 2 0 002 2zm10-10V7a4 4 0 00-8 0v4h8z" />
                                )}
                            </svg>
                            {isAdmin ? 'Admin' : 'Admin Login'}
                        </button>
                    </div>
                    <div className="mt-4">
                        <span className={`px-4 py-2 rounded-full text-sm font-medium ${isVotingOpen ? 'bg-green-500 text-white' : 'bg-red-500 text-white'}`}>
                            {isVotingOpen ? '‚óè Voting Open' : '‚óè Voting Closed'}
                        </span>
                    </div>
                </div>

                {/* Admin Login Panel */}
                {showAdminLogin && !isAdmin && (
                    <div className="bg-white rounded-lg shadow-xl p-6 mb-6">
                        <h3 className="text-xl font-semibold mb-4">Admin Access</h3>
                        <div className="flex gap-2">
                            <input
                                type="password"
                                placeholder="Enter admin password"
                                value={adminPassword}
                                onChange={(e) => setAdminPassword(e.target.value)}
                                onKeyPress={(e) => e.key === 'Enter' && handleAdminLogin()}
                                className="flex-1 px-4 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-purple-500"
                            />
                            <button
                                onClick={handleAdminLogin}
                                className="px-6 py-2 bg-purple-600 text-white rounded-lg hover:bg-purple-700 transition-colors"
                            >
                                Login
                            </button>
                        </div>
                        <p className="text-sm text-gray-500 mt-2">Default password: admin123</p>
                    </div>
                )}

                {/* Admin Controls */}
                {isAdmin && (
                    <div className="bg-white rounded-lg shadow-xl p-6 mb-6">
                        <h3 className="text-xl font-semibold mb-4">Admin Controls</h3>
                        <div className="flex flex-wrap gap-3 items-center">
                            <button
                                onClick={toggleVoting}
                                className={`px-6 py-3 text-white rounded-lg transition-colors font-medium ${
                                    isVotingOpen 
                                        ? 'bg-red-600 hover:bg-red-700' 
                                        : 'bg-green-600 hover:bg-green-700'
                                }`}
                            >
                                {isVotingOpen ? 'Close Voting' : 'Open Voting'}
                            </button>
                            <div className="text-sm text-gray-600">
                                Male votes: {totalMaleVotes} | Female votes: {totalFemaleVotes}
                            </div>
                        </div>
                    </div>
                )}

                {/* Winners (when voting is closed) */}
                {!isVotingOpen && (
                    <div className="bg-white rounded-lg shadow-xl p-6 mb-6">
                        <h2 className="text-2xl font-bold mb-4 flex items-center gap-2">
                            <svg className="w-6 h-6 text-yellow-500" fill="currentColor" viewBox="0 0 24 24">
                                <path d="M12 2L15.09 8.26L22 9.27L17 14.14L18.18 21.02L12 17.77L5.82 21.02L7 14.14L2 9.27L8.91 8.26L12 2Z"/>
                            </svg>
                            Winners Announcement
                        </h2>
                        <div className="grid md:grid-cols-2 gap-6">
                            <div className="p-6 bg-gradient-to-br from-blue-50 to-blue-100 rounded-lg border-2 border-blue-300">
                                <h3 className="text-xl font-bold text-blue-800 mb-3">üèÜ Best Dressed Male</h3>
                                {getMaleWinner() ? (
                                    <>
                                        <p className="text-3xl font-bold text-blue-900 mb-2">{getMaleWinner()}</p>
                                        <p className="text-blue-700">{maleVotes[getMaleWinner()]} votes</p>
                                    </>
                                ) : (
                                    <p className="text-blue-600">No votes yet</p>
                                )}
                            </div>
                            <div className="p-6 bg-gradient-to-br from-pink-50 to-pink-100 rounded-lg border-2 border-pink-300">
                                <h3 className="text-xl font-bold text-pink-800 mb-3">üèÜ Best Dressed Female</h3>
                                {getFemaleWinner() ? (
                                    <>
                                        <p className="text-3xl font-bold text-pink-900 mb-2">{getFemaleWinner()}</p>
                                        <p className="text-pink-700">{femaleVotes[getFemaleWinner()]} votes</p>
                                    </>
                                ) : (
                                    <p className="text-pink-600">No votes yet</p>
                                )}
                            </div>
                        </div>
                    </div>
                )}

                {/* Voting Interface */}
                {isVotingOpen && (
                    <div className="grid md:grid-cols-2 gap-6">
                        {/* Male Colleagues */}
                        <div className="bg-white rounded-lg shadow-xl p-6">
                            <h2 className="text-2xl font-bold mb-4 text-blue-800">Vote for Best Dressed Male</h2>
                            {maleVote && (
                                <div className="mb-4 p-3 bg-blue-50 border border-blue-200 rounded-lg">
                                    <p className="text-sm text-blue-800">
                                        ‚úì You voted for <strong>{maleVote}</strong>. Click another to change your vote.
                                    </p>
                                </div>
                            )}
                            <div className="space-y-2">
                                {maleColleagues.map((name) => {
                                    const isSelected = maleVote === name;
                                    return (
                                        <button
                                            key={name}
                                            onClick={() => handleMaleVote(name)}
                                            className={`w-full p-3 rounded-lg text-left transition-all ${
                                                isSelected
                                                    ? 'bg-gradient-to-r from-blue-100 to-blue-200 border-2 border-blue-500'
                                                    : 'bg-gradient-to-r from-blue-50 to-indigo-50 border-2 border-blue-200 hover:border-blue-400 hover:shadow-md'
                                            }`}
                                        >
                                            <div className="flex items-center justify-between">
                                                <span className="font-semibold">{name}</span>
                                                {isSelected && <span className="text-blue-600">‚úì</span>}
                                            </div>
                                        </button>
                                    );
                                })}
                            </div>
                        </div>

                        {/* Female Colleagues */}
                        <div className="bg-white rounded-lg shadow-xl p-6">
                            <h2 className="text-2xl font-bold mb-4 text-pink-800">Vote for Best Dressed Female</h2>
                            {femaleVote && (
                                <div className="mb-4 p-3 bg-pink-50 border border-pink-200 rounded-lg">
                                    <p className="text-sm text-pink-800">
                                        ‚úì You voted for <strong>{femaleVote}</strong>. Click another to change your vote.
                                    </p>
                                </div>
                            )}
                            <div className="space-y-2">
                                {femaleColleagues.map((name) => {
                                    const isSelected = femaleVote === name;
                                    return (
                                        <button
                                            key={name}
                                            onClick={() => handleFemaleVote(name)}
                                            className={`w-full p-3 rounded-lg text-left transition-all ${
                                                isSelected
                                                    ? 'bg-gradient-to-r from-pink-100 to-pink-200 border-2 border-pink-500'
                                                    : 'bg-gradient-to-r from-pink-50 to-rose-50 border-2 border-pink-200 hover:border-pink-400 hover:shadow-md'
                                            }`}
                                        >
                                            <div className="flex items-center justify-between">
                                                <span className="font-semibold">{name}</span>
                                                {isSelected && <span className="text-pink-600">‚úì</span>}
                                            </div>
                                        </button>
                                    );
                                })}
                            </div>
                        </div>
                    </div>
                )}

                {/* Detailed Results (Admin only, when voting is closed) */}
                {isAdmin && !isVotingOpen && (
                    <div className="mt-6 grid md:grid-cols-2 gap-6">
                        {/* Male Results */}
                        <div className="bg-white rounded-lg shadow-xl p-6">
                            <h3 className="text-xl font-bold mb-4 text-blue-800">Male Voting Results</h3>
                            <div className="space-y-3">
                                {getSortedVotes(maleVotes).map(([name, votes], index) => {
                                    const percentage = totalMaleVotes > 0 ? (votes / totalMaleVotes * 100).toFixed(1) : 0;
                                    return (
                                        <div key={name}>
                                            <div className="flex justify-between mb-1">
                                                <span className="font-semibold">
                                                    {index === 0 && 'üèÜ '}
                                                    {name}
                                                </span>
                                                <span className="text-gray-700">{votes} ({percentage}%)</span>
                                            </div>
                                            <div className="w-full bg-gray-200 rounded-full h-2">
                                                <div
                                                    className="bg-gradient-to-r from-blue-400 to-blue-600 h-2 rounded-full transition-all"
                                                    style={{ width: `${percentage}%` }}
                                                />
                                            </div>
                                        </div>
                                    );
                                })}
                            </div>
                        </div>

                        {/* Female Results */}
                        <div className="bg-white rounded-lg shadow-xl p-6">
                            <h3 className="text-xl font-bold mb-4 text-pink-800">Female Voting Results</h3>
                            <div className="space-y-3">
                                {getSortedVotes(femaleVotes).map(([name, votes], index) => {
                                    const percentage = totalFemaleVotes > 0 ? (votes / totalFemaleVotes * 100).toFixed(1) : 0;
                                    return (
                                        <div key={name}>
                                            <div className="flex justify-between mb-1">
                                                <span className="font-semibold">
                                                    {index === 0 && 'üèÜ '}
                                                    {name}
                                                </span>
                                                <span className="text-gray-700">{votes} ({percentage}%)</span>
                                            </div>
                                            <div className="w-full bg-gray-200 rounded-full h-2">
                                                <div
                                                    className="bg-gradient-to-r from-pink-400 to-pink-600 h-2 rounded-full transition-all"
                                                    style={{ width: `${percentage}%` }}
                                                />
                                            </div>
                                        </div>
                                    );
                                })}
                            </div>
                        </div>
                    </div>
                )}
            </div>
        );
    };

    ReactDOM.render(<BestDressAward />, document.getElementById('app'));
</script>
